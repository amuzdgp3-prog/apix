## 2. Схема базы данных (DDL)

```sql
-- ==========================================
-- СПРАВОЧНИКИ
-- ==========================================

-- Типы аппаратов
CREATE TABLE machine_types (
  id          SERIAL PRIMARY KEY,
  name        VARCHAR(100) NOT NULL,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Игрушки
CREATE TABLE toys (
  id          SERIAL PRIMARY KEY,
  name        VARCHAR(100) NOT NULL,
  price       NUMERIC(10,2) NOT NULL,          -- текущая закупочная цена
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- История цен на игрушки
CREATE TABLE toy_price_history (
  id          SERIAL PRIMARY KEY,
  toy_id      INT NOT NULL REFERENCES toys(id),
  price       NUMERIC(10,2) NOT NULL,
  valid_from  TIMESTAMPTZ NOT NULL DEFAULT now(),
  created_by  INT REFERENCES staff(id)
);

-- Игрушки по типу аппарата (базовый набор)
CREATE TABLE machine_type_toys (
  machine_type_id  INT NOT NULL REFERENCES machine_types(id) ON DELETE CASCADE,
  toy_id           INT NOT NULL REFERENCES toys(id) ON DELETE CASCADE,
  PRIMARY KEY (machine_type_id, toy_id)
);

-- Сотрудники
CREATE TABLE staff (
  id          SERIAL PRIMARY KEY,
  email       VARCHAR(255) NOT NULL UNIQUE,
  full_name   VARCHAR(255) NOT NULL,
  password    VARCHAR(255) NOT NULL,            -- bcrypt
  role        VARCHAR(20) NOT NULL CHECK (role IN ('technician', 'admin')),
  city_id     INT REFERENCES locations(id),     -- для техников
  is_active   BOOLEAN NOT NULL DEFAULT true,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Refresh токены
CREATE TABLE refresh_tokens (
  id          SERIAL PRIMARY KEY,
  staff_id    INT NOT NULL REFERENCES staff(id) ON DELETE CASCADE,
  token_hash  VARCHAR(255) NOT NULL UNIQUE,
  expires_at  TIMESTAMPTZ NOT NULL,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  revoked_at  TIMESTAMPTZ
);

-- ==========================================
-- ЛОКАЦИИ (дерево произвольной глубины)
-- ==========================================

CREATE TABLE locations (
  id               SERIAL PRIMARY KEY,
  name             VARCHAR(255) NOT NULL,
  parent_id        INT REFERENCES locations(id),
  node_type        VARCHAR(50),                 -- 'city', 'district', 'mall', 'point' и т.д.
  min_service_days INT,                         -- интервал обслуживания (наследуется)
  max_service_days INT,
  created_at       TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at       TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Маршруты
CREATE TABLE routes (
  id               SERIAL PRIMARY KEY,
  name             VARCHAR(255) NOT NULL,
  description      TEXT,
  min_service_days INT,
  max_service_days INT,
  created_at       TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at       TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Точки маршрута (упорядоченный список)
CREATE TABLE route_locations (
  route_id     INT NOT NULL REFERENCES routes(id) ON DELETE CASCADE,
  location_id  INT NOT NULL REFERENCES locations(id) ON DELETE CASCADE,
  sort_order   INT NOT NULL DEFAULT 0,
  PRIMARY KEY (route_id, location_id)
);

-- ==========================================
-- АППАРАТЫ
-- ==========================================

CREATE TABLE machines (
  number              INT PRIMARY KEY,           -- номер на корпусе
  location_id         INT NOT NULL REFERENCES locations(id),
  type_id             INT NOT NULL REFERENCES machine_types(id),
  status              VARCHAR(20) NOT NULL DEFAULT 'active' CHECK (status IN ('active', 'inactive')),
  price_per_game      NUMERIC(10,2) NOT NULL,
  has_prize_counter   BOOLEAN NOT NULL DEFAULT true,
  min_service_days    INT,                        -- переопределение интервала
  max_service_days    INT,
  deleted_at          TIMESTAMPTZ,
  created_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at          TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Индивидуальные игрушки аппарата (дополнение/исключение от типа)
CREATE TABLE machine_toys (
  machine_number  INT NOT NULL REFERENCES machines(number) ON DELETE CASCADE,
  toy_id          INT NOT NULL REFERENCES toys(id) ON DELETE CASCADE,
  action          VARCHAR(10) NOT NULL CHECK (action IN ('add', 'remove')), -- добавить или убрать из базового набора
  PRIMARY KEY (machine_number, toy_id)
);

-- Привязка аппаратов к маршрутам
CREATE TABLE machine_routes (
  machine_number  INT NOT NULL REFERENCES machines(number) ON DELETE CASCADE,
  route_id        INT NOT NULL REFERENCES routes(id) ON DELETE CASCADE,
  PRIMARY KEY (machine_number, route_id)
);

-- Привязка техников к аппаратам
CREATE TABLE machine_technicians (
  machine_number  INT NOT NULL REFERENCES machines(number) ON DELETE CASCADE,
  staff_id        INT NOT NULL REFERENCES staff(id) ON DELETE CASCADE,
  PRIMARY KEY (machine_number, staff_id)
);

-- ==========================================
-- СЕССИИ (placements)
-- ==========================================

-- Сессия = один физический аппарат на одной точке
CREATE TABLE machine_placements (
  id                    SERIAL PRIMARY KEY,
  machine_number        INT NOT NULL REFERENCES machines(number),
  started_at            TIMESTAMPTZ NOT NULL DEFAULT now(),
  ended_at              TIMESTAMPTZ,              -- NULL = текущая сессия
  game_counter_initial  INT NOT NULL DEFAULT 0,
  prize_counter_initial INT NOT NULL DEFAULT 0,
  created_by            INT REFERENCES staff(id),
  created_at            TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- ==========================================
-- ОБСЛУЖИВАНИЯ
-- ==========================================

CREATE TABLE service (
  id                    SERIAL PRIMARY KEY,
  placement_id          INT NOT NULL REFERENCES machine_placements(id),
  machine_number        INT REFERENCES machines(number),  -- NULL если аппарат удалён
  staff_id              INT NOT NULL REFERENCES staff(id),
  service_date          DATE NOT NULL,
  service_time          TIME NOT NULL,
  game_counter          INT NOT NULL,
  prize_counter         INT,
  test_games            INT NOT NULL DEFAULT 0,
  is_operational        BOOLEAN NOT NULL DEFAULT true,    -- работает/не работает
  comment               TEXT,

  -- Снапшоты цен на момент обслуживания
  price_per_game        NUMERIC(10,2) NOT NULL,

  -- Вычисляемые поля (заполняются сервером при пересчёте цепочки)
  before_service_id     INT REFERENCES service(id),
  period_days           INT,
  new_games             INT,
  new_prizes            INT,
  revenue               NUMERIC(12,2),
  cost_of_toys          NUMERIC(12,2),
  roi                   NUMERIC(8,4),

  -- Атрибуция выручки периода (см. раздел 6 "Аналитика по технику")
  -- revenue_attribution_staff_id = staff_id ПРЕДЫДУЩЕГО обслуживания (того, кто заложил
  -- игрушки, отработавшие этот период). Денормализовано для скорости отчётов, чтобы не
  -- джойнить service на себя каждый раз. Заполняется в chainRecalc.ts вместе с остальными
  -- вычисляемыми полями.
  revenue_attribution_staff_id INT REFERENCES staff(id),
  revenue_per_day        NUMERIC(12,2),   -- revenue / period_days, для честного сравнения периодов разной длины
  is_mixed_period         BOOLEAN,        -- true если prev.staff_id != current.staff_id (эстафета между техниками)

  -- Фото
  photo_counter_url     TEXT,
  photo_before_url      TEXT,
  photo_after_url       TEXT,

  synced_at             TIMESTAMPTZ NOT NULL DEFAULT now(),
  created_at            TIMESTAMPTZ NOT NULL DEFAULT now(),

  UNIQUE (placement_id, service_date)              -- одно обслуживание в день на placement
);

-- Расход игрушек по обслуживанию
CREATE TABLE toy_distribution (
  id              SERIAL PRIMARY KEY,
  service_id      INT NOT NULL REFERENCES service(id) ON DELETE CASCADE,
  toy_id          INT NOT NULL REFERENCES toys(id),
  quantity        INT NOT NULL DEFAULT 0,
  price_snapshot  NUMERIC(10,2) NOT NULL,          -- цена на момент обслуживания
  total_cost      NUMERIC(10,2) GENERATED ALWAYS AS (quantity * price_snapshot) STORED
);

-- ==========================================
-- АУДИТ
-- ==========================================

CREATE TABLE audit_log (
  id           BIGSERIAL PRIMARY KEY,
  table_name   VARCHAR(100) NOT NULL,
  record_id    TEXT NOT NULL,
  action       VARCHAR(10) NOT NULL CHECK (action IN ('INSERT', 'UPDATE', 'DELETE')),
  user_id      INT REFERENCES staff(id),
  changed_at   TIMESTAMPTZ NOT NULL DEFAULT now(),
  old_data     JSONB,
  new_data     JSONB
);

CREATE TABLE audit_log_archive (
  LIKE audit_log INCLUDING ALL
);

-- Индексы
CREATE INDEX idx_service_placement_date ON service(placement_id, service_date);
CREATE INDEX idx_service_date ON service(service_date);
CREATE INDEX idx_service_machine ON service(machine_number);
CREATE INDEX idx_service_staff ON service(staff_id);
CREATE INDEX idx_service_revenue_attribution ON service(revenue_attribution_staff_id);
CREATE INDEX idx_audit_changed_at ON audit_log(changed_at);
CREATE INDEX idx_audit_table_record ON audit_log(table_name, record_id);
CREATE INDEX idx_locations_parent ON locations(parent_id);
CREATE INDEX idx_placements_machine ON machine_placements(machine_number);
```

## 3. Схема IndexedDB (Dexie.js)

```typescript
// packages/client/src/store/db.ts

import Dexie, { Table } from 'dexie';

// Черновик обслуживания
interface DraftService {
  id?: number;                    // autoincrement
  localId: string;                // uuid, для идентификации
  machineNumber: number;
  machineAddress: string;
  serviceDate: string;            // ISO date
  serviceTime: string;            // HH:MM
  gameCounter: number;
  prizeCounter?: number;
  testGames: number;
  isOperational: boolean;
  comment?: string;
  toys: Array<{
    toyId: number;
    toyName: string;
    quantity: number;
    priceSnapshot: number;
  }>;
  photoCounter?: Blob;
  photoBefore?: Blob;
  photoAfter?: Blob;
  status: 'pending' | 'error';
  errors?: string[];
  createdAt: string;
}

// Кэш аппарата
interface CachedMachine {
  number: number;                 // primary key
  address: string;
  locationId: number;
  typeId: number;
  typeName: string;
  status: string;
  pricePerGame: number;
  hasPrizeCounter: boolean;
  minServiceDays?: number;
  maxServiceDays?: number;
  toys: Array<{
    toyId: number;
    name: string;
    price: number;
  }>;
  technicianIds: number[];
  updatedAt: string;
}

// Кэш последних обслуживаний (для тренда ROI и прошлых значений)
interface CachedServiceHistory {
  id?: number;
  machineNumber: number;          // index
  serviceDate: string;
  serviceTime: string;
  gameCounter: number;
  prizeCounter?: number;
  testGames: number;
  newGames?: number;
  revenue?: number;
  costOfToys?: number;
  roi?: number;
  isOperational: boolean;
}

// Кэш локаций
interface CachedLocation {
  id: number;
  name: string;
  parentId?: number;
  nodeType?: string;
  minServiceDays?: number;
  maxServiceDays?: number;
  updatedAt: string;
}

// Кэш игрушек
interface CachedToy {
  id: number;
  name: string;
  price: number;
  updatedAt: string;
}

// Метаданные синхронизации
interface SyncMeta {
  key: string;                    // primary key
  lastSyncAt: string;             // ISO datetime
}

class SlotManagerDB extends Dexie {
  drafts!: Table<DraftService>;
  machines!: Table<CachedMachine>;
  serviceHistory!: Table<CachedServiceHistory>;
  locations!: Table<CachedLocation>;
  toys!: Table<CachedToy>;
  syncMeta!: Table<SyncMeta>;

  constructor() {
    super('SlotManagerDB');
    this.version(1).stores({
      drafts:         '++id, localId, machineNumber, status, createdAt',
      machines:       'number, locationId, typeId, status',
      serviceHistory: '++id, machineNumber, serviceDate',
      locations:      'id, parentId',
      toys:           'id',
      syncMeta:       'key',
    });
  }
}

export const db = new SlotManagerDB();
```

## 4. Спецификация API

### Базовый URL: `/api`
### Аутентификация: Bearer JWT в заголовке `Authorization` (кроме /auth/*)

---

### 4.1. Аутентификация

```
POST /api/auth/login
Body: { email: string, password: string }
Response: { accessToken: string, user: { id, fullName, role, cityId } }
Set-Cookie: refreshToken=... (httpOnly, secure)

POST /api/auth/refresh
Cookie: refreshToken
Response: { accessToken: string }

POST /api/auth/logout
Cookie: refreshToken
Response: { success: true }
```

---

### 4.2. Синхронизация (техник)

```
POST /api/sync
Content-Type: multipart/form-data
Auth: technician | admin

Fields:
  localId        string   (uuid черновика)
  machineNumber  number
  serviceDate    string   (YYYY-MM-DD)
  serviceTime    string   (HH:MM)
  gameCounter    number
  prizeCounter   number?
  testGames      number
  isOperational  boolean
  comment        string?
  toys           string   (JSON: [{toyId, quantity}])
Files:
  photoCounter   File
  photoBefore    File
  photoAfter     File

Response 200:
  { success: true, recordId: number, calculations: {
      newGames, revenue, costOfToys, roi, periodDays
  }}

Response 422:
  { success: false, errors: [
      { field: string, message: string }
  ]}
```

---

### 4.3. Кэш для техника

```
GET /api/cache/machines

Response: {

}
```

## 5. Алгоритм пересчёта цепочки

```typescript
// apps/server/src/services/chainRecalc.ts

async function recalcChain(
  placementId: number,
  startDate: Date,
  tx: Transaction
): Promise<void> {

  // 1. Получить все обслуживания placement начиная с startDate
  const services = await tx
    .select()
    .from(serviceTable)
    .where(
      and(
        eq(serviceTable.placementId, placementId),
        gte(serviceTable.serviceDate, startDate)
      )
    )
    .orderBy(asc(serviceTable.serviceDate), asc(serviceTable.serviceTime))
    .for('update');  // SELECT FOR UPDATE

  // 2. Получить предыдущее обслуживание
  const prevService = await tx
    .select()
    .from(serviceTable)
    .where(
      and(
        eq(serviceTable.placementId, placementId),
        lt(serviceTable.serviceDate, startDate)
      )
    )
    .orderBy(desc(serviceTable.serviceDate), desc(serviceTable.serviceTime))
    .limit(1);

  let prev = prevService[0] ?? null;

  // 3. Пересчёт для каждого обслуживания
  for (const current of services) {
    const updates: Partial<Service> = {};

    updates.beforeServiceId = prev?.id ?? null;

    if (prev) {
      // Период в днях
      const daysDiff = differenceInDays(current.serviceDate, prev.serviceDate);
      updates.periodDays = Math.max(1, daysDiff);

      // Новые игры
      updates.newGames = current.gameCounter - prev.gameCounter - current.testGames;

      // Новые призы
      if (current.prizeCounter != null && prev.prizeCounter != null) {
        updates.newPrizes = current.prizeCounter - prev.prizeCounter;
      }

      // Выручка (только если аппарат работал в предыдущем периоде)
      if (prev.isOperational) {
        updates.revenue = (updates.newGames ?? 0) * current.pricePerGame;
      } else {
        updates.revenue = null; // период простоя
      }

      // ROI = выручка текущего / себестоимость предыдущего
      if (updates.revenue != null && prev.costOfToys && prev.costOfToys > 0) {
        updates.roi = updates.revenue / Number(prev.costOfToys);
      } else {
        updates.roi = null;
      }

      // Атрибуция: выручка периода принадлежит технику, который ЗАЛОЖИЛ игрушки
      // в предыдущее обслуживание — именно его набор игрушек "работал" этот период,
      // независимо от того, кто физически приехал и снял счётчики сейчас.
      updates.revenueAttributionStaffId = prev.staffId;

      // Нормализация на длину периода — иначе долгий период (например, из-за
      // задержки следующего техника) искажает абсолютную сумму не по вине того,
      // кто заложил товар.
      updates.revenuePerDay =
        updates.revenue != null
          ? Number(updates.revenue) / (updates.periodDays ?? 1)
          : null;

      // Флаг "эстафеты" — для админских отчётов, чтобы видеть смешанные периоды
      updates.isMixedPeriod = prev.staffId !== current.staffId;
    } else {
      // Первое обслуживание на placement
      updates.periodDays = null;
      updates.newGames = null;
      updates.newPrizes = null;
      updates.revenue = null;
      updates.roi = null;
      updates.revenueAttributionStaffId = null;
      updates.revenuePerDay = null;
      updates.isMixedPeriod = null;
    }

    await tx
      .update(serviceTable)
      .set(updates)
      .where(eq(serviceTable.id, current.id));

    prev = { ...current, ...updates };
  }
}
```

## 6. Аналитика по технику

Три независимые оси оценки техника. Они отвечают на разные вопросы и **не должны сводиться в одно число** — выручка периода зависит не только от самого техника (см. п. 3.6, 3.8 ТЗ про эстафету между техниками на одном аппарате), поэтому смешивать её с операционными метриками (своевременность, забытые аппараты) методологически неверно.

### 6.1. Влияние закладки техника на выручку

**Принцип атрибуции:** выручка периода `[prev.service_date → current.service_date]` физически — следствие набора игрушек, заложенного в `prev`-обслуживании. Поэтому она атрибутируется **технику предыдущего обслуживания** (`prev.staff_id`), а не тому, кто закрыл период снятием счётчиков. Поле `revenue_attribution_staff_id` заполняется автоматически в `chainRecalc.ts` (см. раздел 5) и денормализовано прямо в `service`, чтобы не джойнить таблицу на себя в каждом отчёте.

Выручка нормализуется на `period_days` (`revenue_per_day`), чтобы случайно длинный или короткий период (например, из-за задержки следующего техника) не искажал оценку — техник, заложивший игрушки, не виноват в том, что за ним приехали с опозданием.

```typescript
// apps/server/src/services/reports/technicianRevenueAttribution.ts

async function getRevenueAttributionStats(
  staffId: number,
  dateFrom: Date,
  dateTo: Date,
  db: Database
) {
  const periods = await db
    .select({
      machineNumber: serviceTable.machineNumber,
      revenue: serviceTable.revenue,
      revenuePerDay: serviceTable.revenuePerDay,
      roi: serviceTable.roi,
      costOfToys: serviceTable.costOfToys,   // себестоимость ЕГО закладки (из prev)
      newGames: serviceTable.newGames,        // новых игр СЛЕДУЮЩЕГО периода
      periodDays: serviceTable.periodDays,
      isMixedPeriod: serviceTable.isMixedPeriod,
      serviceDate: serviceTable.serviceDate,
    })
    .from(serviceTable)
    .where(
      and(
        eq(serviceTable.revenueAttributionStaffId, staffId),
        gte(serviceTable.serviceDate, dateFrom),
        lte(serviceTable.serviceDate, dateTo),
        isNotNull(serviceTable.revenue)        // исключаем периоды простоя (is_operational=false у prev)
      )
    );

  const withRoi = periods.filter((p) => p.roi != null);
  const mixed = periods.filter((p) => p.isMixedPeriod);

  // Себестоимость на одну новую игру следующего периода — показывает,
  // не перекладывает ли техник лишних игрушек без отдачи в виде новых игр
  const costPerGame = periods
    .filter((p) => p.costOfToys != null && p.newGames && p.newGames > 0)
    .map((p) => Number(p.costOfToys) / p.newGames!);

  return {
    periodsAsToysProviderCount: periods.length,
    mixedPeriodsCount: mixed.length,
    avgRoiAsToysProvider: withRoi.length
      ? avg(withRoi.map((p) => Number(p.roi)))
      : null,
    avgRevenuePerDayAsToysProvider: avg(
      periods.map((p) => Number(p.revenuePerDay ?? 0))
    ),
    avgCostPerGame: costPerGame.length ? avg(costPerGame) : null,
  };
}
```

**Сравнение техников между собой (A/B, для спорных ситуаций).** Если нужно честно сравнить двух техников на одних и тех же точках без влияния чужих решений, используется второй, более строгий срез — только периоды, где `prev.staff_id = current.staff_id` (техник полностью владел циклом от своей закладки до своего же следующего визита):

```typescript
async function getCleanPeriodsComparison(
  staffIds: number[],
  dateFrom: Date,
  dateTo: Date,
  db: Database
) {
  return db
    .select({
      staffId: serviceTable.revenueAttributionStaffId,
      avgRoi: avg(serviceTable.roi),
      avgRevenuePerDay: avg(serviceTable.revenuePerDay),
      count: count(),
    })
    .from(serviceTable)
    .where(
      and(
        inArray(serviceTable.revenueAttributionStaffId, staffIds),
        eq(serviceTable.isMixedPeriod, false),   // только "чистые" периоды
        gte(serviceTable.serviceDate, dateFrom),
        lte(serviceTable.serviceDate, dateTo)
      )
    )
    .groupBy(serviceTable.revenueAttributionStaffId);
}
```

### 6.2. Своевременность обслуживания

Атрибуция простая — это про собственный визит техника, без оглядки на соседние обслуживания. Источник интервалов: наследование `район → маршрут → аппарат` (`min_service_days` / `max_service_days`), уже описанное в п. 3.9 ТЗ.

```typescript
// apps/server/src/services/reports/technicianTimeliness.ts

async function getTimelinessStats(
  staffId: number,
  dateFrom: Date,
  dateTo: Date,
  db: Database
) {
  // effectiveMaxDays/effectiveMinDays вычисляются той же функцией наследования,
  // что используется для баннера "забытые аппараты" (useForgottenMachines.ts /
  // forgottenCheck.ts на сервере) — переиспользуем, чтобы логика не разошлась.
  const visits = await db
    .select({
      serviceId: serviceTable.id,
      machineNumber: serviceTable.machineNumber,
      serviceDate: serviceTable.serviceDate,
      periodDays: serviceTable.periodDays,
      effectiveMaxDays: resolveEffectiveMaxDays(serviceTable.machineNumber), // helper, см. forgottenCheck.ts
      effectiveMinDays: resolveEffectiveMinDays(serviceTable.machineNumber),
    })
    .from(serviceTable)
    .where(
      and(
        eq(serviceTable.staffId, staffId),       // визит самого техника, не атрибуция
        gte(serviceTable.serviceDate, dateFrom),
        lte(serviceTable.serviceDate, dateTo),
        isNotNull(serviceTable.periodDays)        // исключаем первое обслуживание на placement
      )
    );

  let onTime = 0, late = 0, early = 0;
  const overdueDays: number[] = [];

  for (const v of visits) {
    if (v.periodDays! > v.effectiveMaxDays) {
      late++;
      overdueDays.push(v.periodDays! - v.effectiveMaxDays);
    } else if (v.periodDays! < v.effectiveMinDays) {
      early++;
    } else {
      onTime++;
    }
  }

  const total = visits.length;
  return {
    onTimePercent: total ? (onTime / total) * 100 : null,
    lateVisitsCount: late,
    earlyVisitsCount: early,
    avgOverdueDays: overdueDays.length ? avg(overdueDays) : 0,
  };
}
```

### 6.3. Забытые аппараты

Два разных временных среза — снэпшот "на сейчас" (для баннера и дашборда) и историческая частота "за период" (для KPI). Это принципиально разные вещи: снэпшот может быть нулевым у техника, который весь месяц допускал просрочки, но к моменту отчёта их уже закрыл другой человек.

```typescript
// apps/server/src/jobs/forgottenCheck.ts (переиспользуется и для отчёта, и для баннера)

// 3a. Снэпшот на сейчас — по аппаратам, закреплённым за техником
async function getCurrentForgottenCount(staffId: number, db: Database) {
  const rows = await db
    .select({ machineNumber: machinesTable.number })
    .from(machinesTable)
    .innerJoin(
      machineTechniciansTable,
      eq(machineTechniciansTable.machineNumber, machinesTable.number)
    )
    .innerJoin(
      lastServiceSubquery, // LATERAL: последнее service по machine_number
      eq(lastServiceSubquery.machineNumber, machinesTable.number)
    )
    .where(
      and(
        eq(machineTechniciansTable.staffId, staffId),
        isNull(machinesTable.deletedAt),
        sql`CURRENT_DATE - ${lastServiceSubquery.serviceDate} > ${resolveEffectiveMaxDays(machinesTable.number)}`
      )
    );

  return rows.length;
}

// 3b. Историческая частота — сколько периодов ПОСЛЕ визита этого техника
// превысили max_days, прежде чем аппарат обслужили снова (или забытость
// зафиксирована до сих пор, current_date - service_date > max_days и нет next).
async function getForgottenHistoryStats(
  staffId: number,
  dateFrom: Date,
  dateTo: Date,
  db: Database
) {
  const periods = await db
    .select({
      periodDays: serviceTable.periodDays,
      effectiveMaxDays: resolveEffectiveMaxDays(serviceTable.machineNumber),
    })
    .from(serviceTable)
    .where(
      and(
        eq(serviceTable.revenueAttributionStaffId, staffId), // "после его визита"
        gte(serviceTable.serviceDate, dateFrom),
        lte(serviceTable.serviceDate, dateTo),
        isNotNull(serviceTable.periodDays)
      )
    );

  const overdue = periods.filter((p) => p.periodDays! > p.effectiveMaxDays);

  return {
    periodsBecameForgottenCount: overdue.length,
    avgDaysOverdueWhenBecameForgotten: overdue.length
      ? avg(overdue.map((p) => p.periodDays! - p.effectiveMaxDays))
      : 0,
  };
}
```

**Важная оговорка про общие маршруты.** Если аппарат закреплён за несколькими техниками одновременно (`machine_technicians` — many-to-many), "забытость" чаще отражает провал координации между ними, а не вину одного человека. Поэтому `periodsBecameForgottenCount` в личном отчёте техника стоит сопровождать пометкой, сколько из этих случаев пришлось на аппараты с более чем одним закреплённым техником (`is_shared_machine`), и в первую очередь показывать "забытые случаи" в разрезе **маршрута** (`/api/reports/locations`), а не только персонально.

```typescript
// Помечаем периоды на "общих" аппаратах для корректной интерпретации в UI

  SELECT COUNT(*) > 1 FROM machine_technicians mt
```

## 7. Стратегии Workbox (Service Worker)

| Маршрут | Стратегия | Описание |
|---|---|---|
| `/` и статические ассеты | CacheFirst | Оболочка PWA всегда из кэша |
| `/api/cache/*` | NetworkFirst | Справочники — сначала сеть, при ошибке кэш |
| `/api/sync` | NetworkOnly | Только сеть, оффлайн — в очередь |
